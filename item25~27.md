```javascript
function fetchPagesCB() {
  let numDone = 0;
  const responses: string[] = [];
  const done = () => {
    const [response1, response2, response3] = responses;
    // ...
  };
  const urls = [url1, url2, url3];
  urls.forEach((url, i) => {
    fetchURL(url, r => {
      responses[i] = url;
      numDone++;
      if (numDone === urls.length) done();
    });
  });
}



async function fetchPages() {
  const [response1, response2, response3] = await Promise.all([
    fetch(url1), fetch(url2), fetch(url3)
  ]);
  // ...
}

```

콜백으로  작성된 코드는, 
- 작동하는 순서를 예측하기가 어렵다.
- 코드가 길어지며 가독성이 떨어진다.
- 가독성이 떨어지면 타입추론도 어려워진다.


```javascript
// Promis.race()
function timeout(millis: number): Promise<never> {
  return new Promise((resolve, reject) => {
     setTimeout(() => reject('timeout'), millis);
  });
}

async function fetchWithTimeout(url: string, ms: number) {
  return Promise.race([fetch(url), timeout(ms)]);
}

```

# 타입추론에 문맥이 어떻게 사용되는지 이해하기

> 2줄요약
1. 문맥을 잃을시, 타입체커는 타입추론에 실패하며 에러가 난다.
2. 변수로 뽑아서 선언하는 경우, 타입 선언을 추가해줘야한다.

## 문자열타입과 문자열 리터럴타입의 차이

```javascript
function setLanguage(language: string) { /* ... */ }

setLanguage('JavaScript');  // OK

let language = 'JavaScript';
setLanguage(language);  // OK

/*----------------------------------------------------------------*/

type Language = 'JavaScript' | 'TypeScript' | 'Python';
function setLanguage(language: Language) { /* ... */ }

setLanguage('JavaScript');  // OK

let language = 'JavaScript';
setLanguage(language); //ERROR
// ~~~~~~~~ Argument of type 'string' is not assignable
//          to parameter of type 'Language'


//Solution1
type Language = 'JavaScript' | 'TypeScript' | 'Python';
function setLanguage(language: Language) { /* ... */ }
let language: Language = 'JavaScript';
setLanguage(language);  // OK

//Solution2
const language = 'JavaScript'
setLanguage(language); //OK

```
_**Solution2의 경우 타입체커에게 language를 변경할 수 없다고 알려줌으로써 타입체커는 좀더 정확한 타입인 리터럴 'JavaScript'로 추론하도록 한다.**_

## 튜플타입
```javascript
function panTo(where: [number, number]) { /* ... */ }

panTo([10, 20]);  // OK

const loc = [10, 20];
panTo(loc);
//    ~~~ Argument of type 'number[]' is not assignable to
//        parameter of type '[number, number]'


//Solution1
function panTo(where: [number, number]) { /* ... */ }
const loc: [number, number] = [10, 20];
panTo(loc);  // OK



//Solution2
function panTo(where: readonly [number, number]) { 
/* ... */ }
const loc = [10, 20] as const;
panTo(loc);  // OK

// const loc = [10, 20] as const; 
// panTo(loc);
// as const는 deep하게 상수임을 보장한다.
//readonly [number,number]로 추론해서 과하게 정확하다.

// 파라미터에 readonly를 추가해줘야한다.


```


## 객체에서의 타입
```javascript
type Language = "Javascript" | "Typescript";

  language: Language;
```

## 콜백에서의 타입
```javascript
function callWithRandom(fn: (n1: number, n2: number) => void) {
  fn(Math.random(), Math.random());
}
callWithRandom((a, b) => {
  a; // type: number
  b; // type: number
  console.log(a + b);
});

const fn = (a, b) => {
  console.log(a + b); 
};
callWithRandom(fn);
// noImplicitany 에러! 
//'a', 'b' 매개변수에는 암시적으로 'any' 형식이 포함됩니다.
// 따로 함수를 선언할때, 맥락을 잃는다.


//Solution
const fn = (a: number, b: number) => {
  console.log(a + b);
};
callWithRandom(fn); // 정상
```


# 함수형 기법과 라이브러리로 타입 흐름 유지하기

> 한줄요약
가독성을 높이고, 명시적인 타입구문의 필요성을 줄이기 위해 내장된 함수형 기법과, 로대시 같은 유틸리티 라이브러리를 잘 사용하자!
**_절차 < 함수형 < 라이브러리_**


csv파일을 파싱하는 함수를 작성해보자.
## 절차형 프로그래밍
```javascript
const csvData = "...";
const rawRows = csvData.split('\n');
const headers = rawRows[0].split(',');

const rows = rawRows.slice(1).map(rowStr => {
  const row = {};
  rowStr.split(',').forEach((val, j) => {
    row[headers[j]] = val;
  });
  return row;
});
```

## 함수형 프로그래밍
```javascript

const rows = rawRows.slice(1)
					.map(rowStr => rowStr.split(',').reduce((row,val,i)=>(row[headers[i]] = val,row),{}))
});
```

## 라이브러리 사용
```javascript
// requires node modules: @types/lodash

const csvData = "...";
const rawRows = csvData.split('\n');
const headers = rawRows[0].split(',');
import _ from 'lodash';
const rows = rawRows.slice(1)
    .map(rowStr => _.zipObject(headers, rowStr.split(',')));
```

## 다른 예시
```javascript
// requires node modules: @types/lodash

const csvData = "...";
const rawRows = csvData.split('\n');
const headers = rawRows[0].split(',');
import _ from 'lodash';
interface BasketballPlayer {
  name: string;
  team: string;
  salary: number;
}
declare const rosters: {[team: string]: BasketballPlayer[]};

//1.
let allPlayers: BasketballPlayer[] = [];
for (const players of Object.values(rosters)) {
  allPlayers = allPlayers.concat(players);  // OK
}

//2. 내장함수 flat을 사용하면 알아서 타입추론!
const allPlayers = Object.values(rosters).flat();


// 팀내 최고연봉자를 콘솔에 출력하는 함수.
//1. 라이브러리 없을 때,
const teamToPlayers: {[team: string]: BasketballPlayer[]} = {};
for (const player of allPlayers) {
  const {team} = player;
  teamToPlayers[team] = teamToPlayers[team] || [];
  teamToPlayers[team].push(player);
}

for (const players of Object.values(teamToPlayers)) {
  players.sort((a, b) => b.salary - a.salary);
}

const bestPaid = Object.values(teamToPlayers).map(players => players[0]);
bestPaid.sort((playerA, playerB) => playerB.salary - playerA.salary);
console.log(bestPaid);

//2. 로대쉬

const allPlayers = Object.values(rosters).flat();
// OK, type is BasketballPlayer[]
const bestPaid = _(allPlayers)
  .groupBy(player => player.team)
  .mapValues(players => _.maxBy(players, p => p.salary)!)
  .values()
  .sortBy(p => -p.salary)
  .value()  // Type is BasketballPlayer[]


```
